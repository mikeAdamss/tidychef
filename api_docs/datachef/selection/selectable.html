 36def _reverse_direction(direction: Direction):
 37    """
 38    Helper to reverse the provided direction to help make
 39    the api conceptually easier for users.
 40    """
 41
 42    if direction.name in ["down", "below"]:
 43        return up
 44    elif direction.name in ["up", "above"]:
 45        return down
 46    elif direction.name == "left":
 47        return right
 48    elif direction.name == "right":
 49        return left
 50    else:
 51        raise Exception(f"Unable to identify direction: {direction}")

108    @dontmutate
109    def expand(self, direction: Direction):
110        """
111        Given a direction of UP, DOWN, LEFT, RIGHT
112        Expands the current selection of cells in that direction.
113
114        Notes:
115        - Will also accept ABOVE and BELOW as direction, as they
116        are aliases of UP and DOWN respectively.
117        """
118        direction._confirm_pristine()
119        selection: List[BaseCell] = []
120
121        # To begin with, the potential cells is equal to all cells
122        # not currently selected.
123        potential_cells: List[Cell] = dfc.cells_not_in(self.pcells, self.cells)
124
125        if direction in [up, down, above, below]:
126
127            # Only consider things on the same horizontal(x) index as a cell
128            # that's already selected.
129            all_used_x_indicies: FrozenSet[int] = set(c.x for c in self.cells)
130
131            # Now consider each relevant x index
132            for xi in all_used_x_indicies:
133
134                # All cells on this index (i.e in this column)
135                selected_cells_on_xi = dfc.cells_on_x_index(self.cells, xi)
136
137                # Not currently selected cells on this index
138                potential_cells_on_xi: List[Cell] = [
139                    c for c in potential_cells if c.x == xi
140                ]
141
142                if direction in [up, above]:
143                    # Note: A vertical index value works in reverse to a visual relationship,
144                    # the higher the index number -> the high the row number -> the LOWER it is
145                    # in visual terms.
146                    largest_used_yi = dfc.minimum_y_offset(selected_cells_on_xi)
147                    # Add cells from the potential selection to the
148                    # actual selection if they meet the criteria.
149                    selection += [
150                        c
151                        for c in potential_cells_on_xi
152                        if c.is_above(largest_used_yi)  # above: visually
153                    ]
154
155                if direction in [down, below]:
156                    # Note: A vertical index value works in reverse to a visual relationship,
157                    # the lower the index number -> the lower the row number -> the HIGHER it is
158                    # in visual terms.
159                    lowest_used_xi = dfc.maximum_y_offset(selected_cells_on_xi)
160                    selection += [
161                        c
162                        for c in potential_cells_on_xi
163                        if c.is_below(lowest_used_xi)  # below: visually
164                    ]
165
166        if direction in [left, right]:
167
168            # For every row in which have at least one cell selected
169            all_used_y_indicies: FrozenSet[int] = set(c.y for c in self.cells)
170            for yi in all_used_y_indicies:
171
172                # Get all currently selected cells on that row
173                selected_cells_on_yi = dfc.cells_on_y_index(self.cells, yi)
174
175                # Get all not selected cells on that row
176                potential_cells_on_yi: List[Cell] = [
177                    c for c in potential_cells if c.y == yi
178                ]
179
180                if direction == left:
181
182                    # Select anything to the left of the
183                    # rightmost of the selected cells on this row
184                    leftmost_used_yi = dfc.minimum_x_offset(selected_cells_on_yi)
185                    selection += [
186                        c
187                        for c in potential_cells_on_yi
188                        if c.is_left_of(leftmost_used_yi)
189                    ]
190
191                if direction == right:
192                    rightmost_used_yi = dfc.maximum_x_offset(selected_cells_on_yi)
193                    selection += [
194                        c
195                        for c in potential_cells_on_yi
196                        if c.is_right_of(rightmost_used_yi)
197                    ]
198
199        self.cells += selection
200        return self

202    @dontmutate
203    def fill(self, direction: Direction):
204        """
205        Creates a new selection from the cells in that direction
206        relative to the current cell selection.
207
208        :direction: One of: up, down, left, right
209        """
210        did_have = copy.deepcopy(self.cells)
211        self = self.expand(direction)
212        self.cells = [x for x in self.cells if x not in did_have]
213        return self

215    @dontmutate
216    def shift(
217        self,
218        direction_or_x: Union[Direction, int],
219        possibly_y: Optional[int] = None,
220    ):
221        """
222        Move the entire current selection relatively, examples:
223
224        - .shift(right)
225        - .shift(right(5))
226        - .shift(2, 6)
227        - .shift(-1, 4)
228
229        :param direction_or_x: Either a direction of the raw x offset
230        of a direction.
231        :param possibly_y: Either none or the raw y offset of a direction
232        """
233
234        if isinstance(direction_or_x, int):
235            if not isinstance(possibly_y, int):
236                raise BadShiftParameterError()
237            x_offset = direction_or_x
238            y_offset = possibly_y
239        elif isinstance(direction_or_x, BaseDirection):
240            assert (
241                not possibly_y
242            ), "Where passing a direction into shift, that must be the only argument"
243            x_offset = direction_or_x.x
244            y_offset = direction_or_x.y
245        else:
246            raise BadShiftParameterError()
247
248        wanted_cells: List[BaseCell] = [
249            BaseCell(x=c.x + x_offset, y=c.y + y_offset) for c in self.cells
250        ]
251
252        found_cells = dfc.matching_xy_cells(self.pcells, wanted_cells)
253
254        if len(found_cells) == 0 and len(wanted_cells) > 0:
255            raise OutOfBoundsError(
256                "You are attempting to shift your selection "
257                "entirely outside of the boundary of the table."
258            )
259
260        self.cells = found_cells
261        return self

263    @dontmutate
264    def excel_ref(self, excel_ref: str):
265        """
266        Selects just the cells as indicated by the provided excel style
267        reference: "A6", "B17:B24", "9", "GH" etc.
268        """
269
270        # Multi excel reference:
271        # eg: 'B2:F5'
272        if re.match("^[A-Z]+[0-9]+:[A-Z]+[0-9]+$", excel_ref):
273            wanted: List[BaseCell] = dfc.multi_excel_ref_to_basecells(excel_ref)
274            selected = dfc.exactly_matched_xy_cells(self.cells, wanted)
275            # TODO - guarantee sensible ordering?
276
277        # Single column and row reference
278        # eg: 'F19'
279        elif re.match("^[A-Z]+[0-9]+$", excel_ref):
280            wanted: BaseCell = dfc.single_excel_ref_to_basecell(excel_ref)
281            wanted = [wanted]
282            selected = dfc.exactly_matched_xy_cells(self.cells, wanted)
283
284        # An excel reference that is a single row number
285        # eg: '4'
286        elif re.match("^[0-9]+$", excel_ref):
287            wanted_y_index: int = dfc.single_excel_row_to_y_index(excel_ref)
288            selected = [c for c in self.cells if c.y == wanted_y_index]
289
290        # An excel reference that is a multiple row numbers
291        # eg: '4:6'
292        elif re.match("^[0-9]+:[0-9]+$", excel_ref):
293            start_y = excel_ref.split(":")[0]
294            end_y = excel_ref.split(":")[1]
295            start_y_index: int = dfc.single_excel_row_to_y_index(start_y)
296            end_y_index: int = dfc.single_excel_row_to_y_index(end_y)
297            if start_y_index >= end_y_index:
298                raise BadExcelReferenceError(
299                    f'Excel ref "{excel_ref}" is invalid. {end_y_index} must be higher than {start_y_index}'
300                )
301            selected = [
302                c for c in self.cells if c.y >= start_y_index and c.y <= end_y_index
303            ]
304
305        # An excel reference that is one column letter
306        # eg: 'H'
307        elif re.match("^[A-Z]+$", excel_ref):
308            wanted_x_index: int = dfc.single_excel_column_to_x_index(excel_ref)
309            selected = [c for c in self.cells if c.x == wanted_x_index]
310
311        # An excel reference that is a range of column letters
312        # eg: 'H:J'
313        elif re.match("^[A-Z]+:[A-Z]+$", excel_ref):
314            left_letters = excel_ref.split(":")[0]
315            right_letters = excel_ref.split(":")[1]
316            start_x_index: int = dfc.single_excel_column_to_x_index(left_letters)
317            end_x_index: int = dfc.single_excel_column_to_x_index(right_letters)
318            if start_x_index >= end_x_index:
319                raise BadExcelReferenceError(
320                    f'Excel ref "{excel_ref}" is invalid. {right_letters} much be higher than {left_letters}'
321                )
322            selected = [
323                c for c in self.cells if c.x >= start_x_index and c.x <= end_x_index
324            ]
325
326        # Unknown excel reference
327        else:
328            raise BadExcelReferenceError(f"Unrecognised excel reference {excel_ref}")
329
330        self.cells = selected
331        return self

333    def validate(
334        self,
335        validator: BaseValidator,
336        raise_first_error: bool = False
337    ):
338        """
339        Validates current cell selection by passing each currently
340        selected cell to the provided validator.
341
342        Pass raise_first_error=True if you just want the first
343        invalid value message.
344        """
345
346        validation_errors = []
347        for cell in self.cells:
348            if not validator(cell):
349                if raise_first_error:
350                    raise CellValidationError(validator.msg(cell))
351                else:
352                    validation_errors.append(validator.msg(cell))
353                
354        if len(validation_errors) > 0:
355            raise CellValidationError(f"{linesep}".join(validation_errors))
356
357        return self

383    def finds_observations_directly(self, direction: Direction) -> Directly:
384        """
385        Creates and returns a class:Directly lookup engine
386        that can resolve the correct cell from this selection
387        relative to any given observation.
388        """
389
390        if not self._label:
391            raise MissingLabelError(
392                """
393                You are trying to create a lookup engine for a selection of
394                cells using the .resolve_observations_directly() method but
395                have not yet assigned a label to said selection of cells.
396
397                Please use the .label_as() method to assign a label before
398                attempting this.
399            """
400            )
401
402        # The constructor we provide to the user advertises that the column
403        # lookup engines "finds the observations" but this is purely a
404        # conceptual trick to make a more user friendly api.
405        # In reality that's exactly backwards, an observation actually finds
406        # a column value (by being passed to the lookup engine constructed below).
407        # As such we need to reverse the stated direction.
408        return Directly(self.label, self, _reverse_direction(direction))

471        new_end.x = 0 - start.x
472        new_end.y = 0 - start.y
473
474        return Within(
475            self.label, self, _reverse_direction(direction), new_start, new_end

109    @dontmutate
110    def expand(self, direction: Direction):
111        """
112        Given a direction of UP, DOWN, LEFT, RIGHT
113        Expands the current selection of cells in that direction.
114
115        Notes:
116        - Will also accept ABOVE and BELOW as direction, as they
117        are aliases of UP and DOWN respectively.
118        """
119        direction._confirm_pristine()
120        selection: List[BaseCell] = []
121
122        # To begin with, the potential cells is equal to all cells
123        # not currently selected.
124        potential_cells: List[Cell] = dfc.cells_not_in(self.pcells, self.cells)
125
126        if direction in [up, down, above, below]:
127
128            # Only consider things on the same horizontal(x) index as a cell
129            # that's already selected.
130            all_used_x_indicies: FrozenSet[int] = set(c.x for c in self.cells)
131
132            # Now consider each relevant x index
133            for xi in all_used_x_indicies:
134
135                # All cells on this index (i.e in this column)
136                selected_cells_on_xi = dfc.cells_on_x_index(self.cells, xi)
137
138                # Not currently selected cells on this index
139                potential_cells_on_xi: List[Cell] = [
140                    c for c in potential_cells if c.x == xi
141                ]
142
143                if direction in [up, above]:
144                    # Note: A vertical index value works in reverse to a visual relationship,
145                    # the higher the index number -> the high the row number -> the LOWER it is
146                    # in visual terms.
147                    largest_used_yi = dfc.minimum_y_offset(selected_cells_on_xi)
148                    # Add cells from the potential selection to the
149                    # actual selection if they meet the criteria.
150                    selection += [
151                        c
152                        for c in potential_cells_on_xi
153                        if c.is_above(largest_used_yi)  # above: visually
154                    ]
155
156                if direction in [down, below]:
157                    # Note: A vertical index value works in reverse to a visual relationship,
158                    # the lower the index number -> the lower the row number -> the HIGHER it is
159                    # in visual terms.
160                    lowest_used_xi = dfc.maximum_y_offset(selected_cells_on_xi)
161                    selection += [
162                        c
163                        for c in potential_cells_on_xi
164                        if c.is_below(lowest_used_xi)  # below: visually
165                    ]
166
167        if direction in [left, right]:
168
169            # For every row in which have at least one cell selected
170            all_used_y_indicies: FrozenSet[int] = set(c.y for c in self.cells)
171            for yi in all_used_y_indicies:
172
173                # Get all currently selected cells on that row
174                selected_cells_on_yi = dfc.cells_on_y_index(self.cells, yi)
175
176                # Get all not selected cells on that row
177                potential_cells_on_yi: List[Cell] = [
178                    c for c in potential_cells if c.y == yi
179                ]
180
181                if direction == left:
182
183                    # Select anything to the left of the
184                    # rightmost of the selected cells on this row
185                    leftmost_used_yi = dfc.minimum_x_offset(selected_cells_on_yi)
186                    selection += [
187                        c
188                        for c in potential_cells_on_yi
189                        if c.is_left_of(leftmost_used_yi)
190                    ]
191
192                if direction == right:
193                    rightmost_used_yi = dfc.maximum_x_offset(selected_cells_on_yi)
194                    selection += [
195                        c
196                        for c in potential_cells_on_yi
197                        if c.is_right_of(rightmost_used_yi)
198                    ]
199
200        self.cells += selection
201        return self

461        # conceptual trick to make a more user friendly api.
462        # In reality that's exactly backwards, an observation actually finds
463        # a column value (by being passed to the lookup engine constructed below).
464        # As such we need to reverse the stated direction.
465        # In the case of Within we also need to reverse the direction of
466        # parsing and offsetting.
467        new_start = copy.deepcopy(start)
468        new_start.x = 0 - end.x
469        new_start.y = 0 - end.y
470
471        new_end = copy.deepcopy(end)
472        new_end.x = 0 - start.x
473        new_end.y = 0 - start.y
474
475        return Within(
476            self.label, self, _reverse_direction(direction), new_start, new_end
477        )
