 17class CellRange:
 18    """
 19    A class representing a range of cells
 20
 21    :param low: the lowest index in the range
 22    :param high: the highest index in the range
 23    :param cell: the cell the spanning range resolves to
 24    :direction: the direction of traversal for the lookup
 25    """
 26
 27    low: int
 28    high: int
 29    cell: Cell
 30    direction: Direction
 31
 32    @property
 33    def axis_text(self) -> str:
 34        return "horizontal/x" if self.direction.is_horizontal else "vertical/y"
 35
 36    def contains(self, cell: Cell) -> bool:
 37        """
 38        Does this cell contain the range
 39        in question.
 40
 41        :param cell: The cell we want to check is in the range
 42        """
 43        if self.direction.is_horizontal:
 44            offset = cell.x
 45        else:
 46            offset = cell.y
 47
 48        return offset >= self.low and offset <= self.high
 49
 50    def spans_higher_range_than(self, cell: Cell) -> bool:
 51        """
 52        Does this range cover a range of indexes higher that the
 53        index of a given cell.
 54
 55        :param cell: The cell this range may or may not span a higher
 56        range than.
 57        """
 58        if self.direction.is_horizontal:
 59            return self.low > cell.x
 60        else:
 61            return self.low > cell.y
 62
 63    def spans_lower_range_than(self, cell: Cell) -> bool:
 64        """
 65        Does this range cover a range of indexes lower that the
 66        index of a given cell.
 67
 68        :param cell: The cell this range may or may not span a lower
 69        range than.
 70        """
 71        if self.direction.is_horizontal:
 72            return self.high < cell.x
 73        else:
 74            return self.high < cell.y
 75
 76    def _as_dict(self):
 77        return {
 78            "starts_at": str(self.low),
 79            "ends_at": str(self.high),
 80            "cell": str(self.cell),
 81            "axis": str(self.axis_text),

 85class CellRanges:
 86    """
 87    A class representing multiple cell ranges
 88    """
 89
 90    def __init__(self, selection: LiveTable, direction: Direction, table: str):
 91        """
 92        selection: The selection of cells to create the cell ranges from
 93        :direction: The direction of traversal considered when constructing ranges
 94        """
 95        self.direction: Direction = direction
 96        self.highest_possible_offset: Optional[int] = None
 97        self.lowest_possible_offset: Optional[int] = None
 98        self.table = table
 99        self._populate(selection)
100
101    @property
102    def axis_text(self) -> str:
103        return "horizontal/x" if self.direction.is_horizontal else "vertical/y"
104
105    def get_range_by_index(self, index: int) -> CellRange:
106        """
107        Get a single constructed range by index
108
109        :param index: Index of the range to get
110        """
111        return self.ordered_cell_ranges[index]
112
113    def _populate(self, selection: LiveTable):
114        """
115        Create the cell ranges from the selection provided
116
117        :param selection: The selection of cells.
118        """
119
120        break_points = {}
121        for cell in selection.cells:
122            axis_offset = cell.x if self.direction.is_horizontal else cell.y
123            if axis_offset in break_points.keys():
124                raise AmbiguousLookupError(
125                    f"""
126                    Aborted while processing table "{self.table}.
127                    
128                    You have defined two or more equally valid closest:{self.direction.name}" relationships,
129                    you cannot do this as it creates an ambiguous lookup.
130                    
131                    You are trying to add '{cell}' but we already have: '{break_points[axis_offset]}'.
132                    
133                    Both of these cells have a {self.axis_text} offsets of '{axis_offset}'. 
134                    and so are equally close along this axis from any given observation cell.
135                    """
136                )
137            break_points.update({axis_offset: cell})
138
139        ordered_break_points = [int(k) for k in break_points.keys()]
140        assert len(ordered_break_points) == len(selection.cells)
141        ordered_break_points.sort()
142
143        self.ordered_cell_ranges: Dict[int, CellRange] = {}
144
145        if self.direction.is_left or self.direction.is_upwards:
146            for i in range(0, len(ordered_break_points)):
147
148                low = ordered_break_points[i]
149
150                if i == 0:
151                    self.lowest_possible_offset = low
152
153                if i == len(ordered_break_points) - 1:
154                    high = HIGHEST
155                    self.highest_possible_offset = HIGHEST
156                else:
157                    high = ordered_break_points[i + 1] - 1
158
159                self.ordered_cell_ranges[i] = CellRange(
160                    low=low,
161                    high=high,
162                    cell=break_points[ordered_break_points[i]],
163                    direction=self.direction,
164                )
165
166        else:
167            for i in range(0, len(ordered_break_points)):
168
169                high = ordered_break_points[i]
170
171                if i == len(ordered_break_points) - 1:
172                    self.highest_possible_offset = high
173
174                if i == 0:
175                    low = LOWEST
176                    self.lowest_possible_offset = LOWEST
177                else:
178                    low = ordered_break_points[i - 1] + 1
179
180                self.ordered_cell_ranges[i] = CellRange(
181                    low=low,
182                    high=high,
183                    cell=break_points[ordered_break_points[i]],
184                    direction=self.direction,
185                )
186
187        assert len(self.ordered_cell_ranges) == len(selection.cells)
188
189    def _as_dict(self):
190        """
191        Reproduce the defined ranges as a dict.
192        This functionality is purely for debugging and test purposes.
193        """
194        d = {}
195        cell_range: CellRange
196        for i, cell_range in self.ordered_cell_ranges.items():
197            d.update({str(i): cell_range._as_dict()})
198        return d

201class Closest(BaseLookupEngine):
202    def __init__(
203        self,
204        label: str,
205        selection: LiveTable,
206        direction: Direction,
207        table: str = "Unnamed Table",
208    ):
209        """
210        Creates a lookup engine to column values defined via
211        the closest visual relationship.
212
213        :param label: The label of the column informed
214        by this lookup engine.
215        :param: direction: one of up,down,left,right,above,below
216        :param: Selection: the selection of cells that hold the column values being looked to.
217        :param table: the name of the table data is being extracted from
218        """
219        self.table = table
220        self.direction = direction
221        self.ranges = CellRanges(selection, direction, table)
222        self.bumped = True
223        self.start_index = None
224        self.label = label
225
226    def _resolve_at_lower_range(self, index, cell, ceiling, floor):
227        """
228        Move the index down as as the cell offset was down/less-than the
229        last ranged we looked at then attempt to resolve again.
230
231        :param index: The index of the last range we looked at
232        :param cell: The cell we're trying to find the correct range for
233        :param ceiling: The highest range index to consider
234        :param floor: The lowest range index to consider
235        """
236
237        if self.bumped == False and index != 0:
238            new_index = index - 1
239            self.bumped = True
240        else:
241            potential_range = ceiling - floor
242            new_index = index - int(potential_range / 2)
243            new_index = new_index if new_index != index else new_index - 1
244
245        if new_index < 0:
246            new_index = 0
247
248        return self.resolve(cell, index=new_index, ceiling=index, floor=floor)
249
250    def _resolve_at_higher_range(self, index, cell, ceiling, floor):
251        """
252        Move the index down as as the cell offset was above/greater-than the
253        last ranged we looked at then attempt to resolve again.
254
255        :param index: The index of the last range we looked at
256        :param cell: The cell we're trying to find the correct range for
257        :param ceiling: The lowest range index to consider
258        :param floor: The highest range index to consider
259        """
260
261        if self.bumped == False and index != len(self.ranges.ordered_cell_ranges):
262            new_index = index + 1
263            self.bumped = True
264        else:
265            potential_range = ceiling - floor
266            new_index = index + int(potential_range / 2)
267            new_index = new_index if new_index != index else new_index + 1
268
269        return self.resolve(cell, index=new_index, ceiling=ceiling, floor=index)
270
271    def _confirm_within_bounds(self, cell: Cell):
272        """
273        Raise an exception if we're trying to resolve a lookup for a
274        cell that is out of bounds, making a lookup impossible.
275
276        example: we're trying to resolve a closest relations to the left
277        of a cell that is further to the left than any of the ranges we're
278        considering.
279
280        :param cell: The single Cell object we want to resolve a the relative
281        column cell object for.
282        """
283
284        err_str = """
285                When processing table {table}.
286
287                Lookup for observation cell '{cell}' is impossible. No
288                cells in your selection exist in direction '{direction.name}'
289                relative to this cell.
290
291                The boundary (furthest cell index) out of the cells you provided
292                on the `{axis}` axis has an offset of `{boundary}`.
293                """
294
295        out_of_bounds = [
296            self.direction.is_upwards
297            and cell.is_above(self.ranges.lowest_possible_offset),
298            self.direction.is_downwards
299            and cell.is_below(self.ranges.highest_possible_offset),
300            self.direction.is_left
301            and cell.is_left_of(self.ranges.lowest_possible_offset),
302            self.direction.is_right
303            and cell.is_right_of(self.ranges.highest_possible_offset),
304        ]
305
306        if True in out_of_bounds:
307            axis = "x" if self.direction.is_horizontal else "y"
308            raise ImpossibleLookupError(
309                err_str.format(
310                    table=self.table,
311                    cell=cell,
312                    direction=self.direction,
313                    axis=axis,
314                    boundary=self.ranges.lowest_possible_offset
315                    if any([self.direction.is_upwards, self.direction.is_left])
316                    else self.ranges.highest_possible_offset,
317                )
318            )
319
320    def resolve(self, cell: Cell, index=None, ceiling=None, floor=0) -> Cell:
321        """
322        Given the cell we want to lookup the relative column value for, use
323        a bisection search to identify the correct range in our ordered list of ranges.
324
325        Note - this method is called recursively, using the kwargs to start
326        again at a different point in the indexed list of ranges.
327
328        :param index: The index of the last range we looked at
329        :param cell: The cell we're trying to find the correct range for
330        :param ceiling: The lowest range index to consider
331        :param floor: The highest range index to consider
332        :return: The datachef Cell object representing the
333        column cell.
334        """
335
336        # If neither the index or ceiling have been passed in, then its the
337        # very first lookup - confirm cell is within the bounds and therefore
338        # this lookup is theoretically possible.
339        if not ceiling and not index:
340            self._confirm_within_bounds(cell)
341
342        # If no maximum cell range (ceiling) is set then its the absolute maximum
343        if ceiling is None:
344            ceiling: int = len(self.ranges.ordered_cell_ranges) - 1
345
346        if index is None:
347            # When starting range index is not passed in, start at the mid point.
348            mid_point: int = int(len(self.ranges.ordered_cell_ranges) / 2)
349            index = list(self.ranges.ordered_cell_ranges.keys())[mid_point]
350
351        considered_range: CellRange = self.ranges.get_range_by_index(index)
352
353        if considered_range.spans_lower_range_than(cell):
354            return self._resolve_at_higher_range(
355                index, cell, ceiling=ceiling, floor=floor
356            )
357        elif considered_range.spans_higher_range_than(cell):
358            return self._resolve_at_lower_range(
359                index, cell, ceiling=ceiling, floor=floor
360            )
361        else:
362            # Lookup Caching
363            # --------------
364            # cells are implicitly selected right->down-a-row->right as you look at a tabulated
365            # (i.e as per standard human reading convention) so we'll cache this range index as
366            # there's a significant chance the next obs lookup is in the same range
367            #
368            # this right-then-down-then-right pattern also means that (often, not guaranteed),
369            # if the next obs isn't in the last range checked, there's a decent chance it's in a
370            # neighbouring range, so we'll also always "bump" the index once in the relevant
371            # direction on a first miss.
372            self.index = index
373            self.bumped = False
374
375            return considered_range.cell

258        :param ceiling: The lowest range index to consider
259        :param floor: The highest range index to consider
260        """
261
262        if self.bumped == False and index != len(self.ranges.ordered_cell_ranges):
263            new_index = index + 1
264            self.bumped = True
265        else:
266            potential_range = ceiling - floor
267            new_index = index + int(potential_range / 2)
268            new_index = new_index if new_index != index else new_index + 1
269
270        return self.resolve(cell, index=new_index, ceiling=ceiling, floor=index)
271
272    def _confirm_within_bounds(self, cell: Cell):
273        """
274        Raise an exception if we're trying to resolve a lookup for a
275        cell that is out of bounds, making a lookup impossible.
276
277        example: we're trying to resolve a closest relations to the left
278        of a cell that is further to the left than any of the ranges we're
279        considering.
280
281        :param cell: The single Cell object we want to resolve a the relative
282        column cell object for.
283        """
284
285        err_str = """
286                When processing table {table}.
287
288                Lookup for observation cell '{cell}' is impossible. No
289                cells in your selection exist in direction '{direction.name}'
290                relative to this cell.
291
292                The boundary (furthest cell index) out of the cells you provided
293                on the `{axis}` axis has an offset of `{boundary}`.
294                """
295
296        out_of_bounds = [
297            self.direction.is_upwards
298            and cell.is_above(self.ranges.lowest_possible_offset),
299            self.direction.is_downwards
300            and cell.is_below(self.ranges.highest_possible_offset),
301            self.direction.is_left
302            and cell.is_left_of(self.ranges.lowest_possible_offset),
303            self.direction.is_right
304            and cell.is_right_of(self.ranges.highest_possible_offset),
305        ]
306
307        if True in out_of_bounds:
308            axis = "x" if self.direction.is_horizontal else "y"
309            raise ImpossibleLookupError(
310                err_str.format(
311                    table=self.table,
312                    cell=cell,
313                    direction=self.direction,
314                    axis=axis,
315                    boundary=self.ranges.lowest_possible_offset
316                    if any([self.direction.is_upwards, self.direction.is_left])
317                    else self.ranges.highest_possible_offset,
318                )
319            )
320
321    def resolve(self, cell: Cell, index=None, ceiling=None, floor=0) -> Cell:
322        """
323        Given the cell we want to lookup the relative column value for, use
324        a bisection search to identify the correct range in our ordered list of ranges.
325
326        Note - this method is called recursively, using the kwargs to start
327        again at a different point in the indexed list of ranges.
328
329        :param index: The index of the last range we looked at
330        :param cell: The cell we're trying to find the correct range for
331        :param ceiling: The lowest range index to consider
332        :param floor: The highest range index to consider
333        :return: The datachef Cell object representing the
334        column cell.
335        """
336
337        # If neither the index or ceiling have been passed in, then its the
338        # very first lookup - confirm cell is within the bounds and therefore
339        # this lookup is theoretically possible.
340        if not ceiling and not index:
341            self._confirm_within_bounds(cell)
342
343        # If no maximum cell range (ceiling) is set then its the absolute maximum
344        if ceiling is None:
345            ceiling: int = len(self.ranges.ordered_cell_ranges) - 1
346
347        if index is None:
348            # When starting range index is not passed in, start at the mid point.
349            mid_point: int = int(len(self.ranges.ordered_cell_ranges) / 2)
350            index = list(self.ranges.ordered_cell_ranges.keys())[mid_point]
351
352        considered_range: CellRange = self.ranges.get_range_by_index(index)
353
354        if considered_range.spans_lower_range_than(cell):
355            return self._resolve_at_higher_range(
356                index, cell, ceiling=ceiling, floor=floor
357            )
358        elif considered_range.spans_higher_range_than(cell):
359            return self._resolve_at_lower_range(
360                index, cell, ceiling=ceiling, floor=floor
361            )
362        else:
363            # Lookup Caching
364            # --------------
365            # cells are implicitly selected right->down-a-row->right as you look at a tabulated
366            # (i.e as per standard human reading convention) so we'll cache this range index as
367            # there's a significant chance the next obs lookup is in the same range
368            #
369            # this right-then-down-then-right pattern also means that (often, not guaranteed),
370            # if the next obs isn't in the last range checked, there's a decent chance it's in a
371            # neighbouring range, so we'll also always "bump" the index once in the relevant
372            # direction on a first miss.
373            self.index = index
374            self.bumped = False
375
376            return considered_range.cell
